/**
 * 编写一个应用程序，定义两个整型变量n1,n2并赋给任意值。计算输出n1>n2，n1=0，n1-n2<=0，n1%n2= =0的值。

		int n1 = Integer.parseInt(sNum1);

		String sNum2 = scanner.nextLine();

			ret1 = true;

		}
